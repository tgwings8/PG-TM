'''
MAD LIBS GAME
=============

Below is a simple version of a Mad Lib. Take this example and make your own!
'''


print("Write a noun")
noun1 = input()

print("Write another noun")
noun2 = input()

print("Write an adjective")
adjective1 = input()

print("Write a verb")
verb1 = input()


### MAD LIB ###

print("The " + adjective1 + " " + noun1 + " went to the " + noun2 + " to " + verb1 + ".")
